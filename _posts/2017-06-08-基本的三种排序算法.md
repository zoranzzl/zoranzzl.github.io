### 插入排序
``` C

#include<stdio.h>
int main()
{
   int A[50];
   int i,j,temp,count;
   char ch;
   do
   {
       system("cls");  //清屏函数
       printf("请输入整形数组(要求长度不超过50):\n");
	   i=0;
	   do
       {
           scanf("%d",&A[i]);
           i++;
       }while((ch=getchar())!='\n');   //判断是否输入了回车符
	   count=i;
   }while(count<2);

   printf("\n排序前：\n");
   for(i=0;i<count;i++)
            printf("%d  ",A[i]);

   for(i=1;i<count;i++)
   {
       temp=A[i];
       j=i-1;
       while(j>=0 && A[j]>temp)
       {
           A[j+1]=A[j];
           j--;
       }
      A[j+1]=temp;
   }

    printf("\n排序后：\n");
	for(i=0;i<count;i++)
            printf("%d  ",A[i]);
	return 0;
}
```
